// 在您现有的 GAS 代码基础上添加以下函数 // 登录用户 function loginUser(rawPhone, password) { try { console.log('开始登录用户:', rawPhone); if (!rawPhone || !password) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: '手机号和密码不能为空' })) .setMimeType(ContentService.MimeType.JSON); } // 标准化手机号格式 const phone = rawPhone.toString().replace(/\s+|[()-]/g, '').trim(); // 打开电子表格和工作表 const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: `工作表 "${SHEET_NAME}" 不存在` })) .setMimeType(ContentService.MimeType.JSON); } // 获取所有数据 const lastRow = sheet.getLastRow(); if (lastRow <= 1) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: '数据库为空' })) .setMimeType(ContentService.MimeType.JSON); } // 查找手机号所在行 const phoneColumn = sheet.getRange(2, 1, lastRow - 1, 3).getValues(); // 获取A、B、C三列 for (let i = 0; i < phoneColumn.length; i++) { const dbPhone = String(phoneColumn[i][0]).toString().replace(/\s+|[()-]/g, '').trim(); const dbName = phoneColumn[i][1]; // 姓名在B列 const dbPassword = phoneColumn[i][2]; // 密码在C列 if (dbPhone === phone) { // 检查密码是否正确 if (!dbPassword) { return ContentService .createTextOutput(JSON.stringify({ success: false, error: '该账号尚未设置密码，请先注册' })) .setMimeType(ContentService.MimeType.JSON); } if (dbPassword.toString() === password.toString()) { // 登录成功，返回用户信息 return ContentService .createTextOutput(JSON.stringify({ success: true, message: "登录成功", name: dbName || "用户", phone: phone })) .setMimeType(ContentService.MimeType.JSON); } else { // 密码错误 return ContentService .createTextOutput(JSON.stringify({ success: false, error: '密码错误' })) .setMimeType(ContentService.MimeType.JSON); } } } // 手机号不存在 return ContentService .createTextOutput(JSON.stringify({ success: false, error: '手机号未注册' })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { console.error('登录用户时出错:', error); return ContentService .createTextOutput(JSON.stringify({ success: false, error: '服务器内部错误: ' + error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // 修改 doPost 函数以包含新的 login 功能 function doPost(e) { try { // 解析请求参数 const action = e.parameter.action; const phone = e.parameter.phone; const password = e.parameter.password; console.log('收到请求:', { action, phone, password }); // 根据 action 参数调用不同的函数 switch (action) { case 'checkPhoneNumber': return checkPhoneNumber(phone); case 'registerWithPassword': return registerWithPassword(phone, password); case 'loginUser': return loginUser(phone, password); default: return ContentService .createTextOutput(JSON.stringify({ success: false, error: '未知的操作: ' + action })) .setMimeType(ContentService.MimeType.JSON); } } catch (error) { console.error('处理请求时出错:', error); return ContentService .createTextOutput(JSON.stringify({ success: false, error: '服务器内部错误: ' + error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
