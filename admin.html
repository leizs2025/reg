// 获取所有用户数据 function getAllUsers() { try { var sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME); var range = sheet.getDataRange(); var values = range.getValues(); var users = []; // 假设第1行为标题，从第2行开始是数据 for (var i = 1; i < values.length; i++) { if (values[i][0]) { // 如果手机号不为空 users.push({ phone: values[i][0], name: values[i][1], password: values[i][2] }); } } return {success: true, data: users}; } catch (e) { return {success: false, error: e.toString()}; } } // 更新用户密码 function updatePassword(data) { try { var sheet = SpreadsheetApp.openById(SPREADSHEET_ID).getSheetByName(SHEET_NAME); var range = sheet.getDataRange(); var values = range.getValues(); // 验证密码格式 if (!data.password.match(/^[a-zA-Z]/)) { return {success: false, error: '密码必须以英文字母开头', code: 'INVALID_PASSWORD_FORMAT'}; } // 查找并更新 for (var i = 1; i < values.length; i++) { if (values[i][0] == data.phone) { // 手机号匹配 sheet.getRange(i + 1, 3).setValue(data.password); // 更新密码列（第3列） return {success: true, message: '密码更新成功'}; } } return {success: false, error: '未找到该手机号的用户'}; } catch (e) { return {success: false, error: e.toString()}; } } // 在doPost主函数中添加路由 function doPost(e) { var action = e.parameter.action; if (action === 'getAllUsers') { return ContentService.createTextOutput( JSON.stringify(getAllUsers()) ).setMimeType(ContentService.MimeType.JSON); } else if (action === 'updatePassword') { return ContentService.createTextOutput( JSON.stringify(updatePassword(e.parameter)) ).setMimeType(ContentService.MimeType.JSON); } // ... 其他现有功能 }
